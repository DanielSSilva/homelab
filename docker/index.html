    Example Docker Compose

    # Example compose file structure
version: '3'
services:
  service-name:
    image: image:tag
    container_name: container-name
    ports:
      - "host:container"
    volumes:
      - /host/path:/container/path
    environment:
      - VARIABLE=value
    restart: unless-stopped
    networks:
      - network-name
